function settingsMergeCustomizer(objValue, srcValue, key, object, source) {
  // 同来源: 后者完全替换前者
  if (isSameSource(object, source)) return srcValue;
  // 跨来源数组: 拼接并去重
  if (Array.isArray(objValue)) {
    return [...new Set([...objValue, ...srcValue])];
  }
  // 对象: 递归深合并
  return undefined; // 使用默认合并
}
